Create Required Tables:

CREATE TABLE  IF NOT EXISTS Users (ID INT PRIMARY KEY AUTO_INCREMENT, Email = VARCHAR(100), Passwd = VARCHAR(255), Code = VARCHAR(255), Status = INT);

CREATE TABLE  IF NOT EXISTS Classes (ClassID INT PRIMARY KEY AUTO_INCREMENT, CreatorID = INT, ClassName = VARCHAR(100), DeletionRequest = INT);

CREATE TABLE  IF NOT EXISTS Questions (QuestionID INT PRIMARY KEY AUTO_INCREMENT, ClassID = INT, CreatorID = INT, QuestionType = INT, QuestionField = VARCHAR(500), DeletionRequest = INT);

CREATE TABLE  IF NOT EXISTS QA (QAID INT PRIMARY KEY AUTO_INCREMENT, QuestionID = INT, QANameField = VARCHAR(255), QAFieldValue = VARCHAR(255), QACorrect = INT);

CREATE TABLE  IF NOT EXISTS Test (TestID INT PRIMARY KEY AUTO_INCREMENT, ClassID = INT, CreatorID = INT, TimeOfTest = INT, DeletionRequest = INT);

CREATE TABLE  IF NOT EXISTS TestQuestions (TQID INT PRIMARY KEY AUTO_INCREMENT, TestID = INT, QuestionID = INT, PointValue = INT);

CREATE TABLE  IF NOT EXISTS StudentClasses (ID INT PRIMARY KEY AUTO_INCREMENT, StudentID = INT, ClassID = INT, TeacherID = INT);

CREATE TABLE  IF NOT EXISTS StudentTests (ID INT PRIMARY KEY AUTO_INCREMENT, StudentID = INT, StudentClassID = INT, TestID = INT, Score = INT, StartTime = DATETIME);

CREATE TABLE  IF NOT EXISTS StudentTestQuestions (ID INT PRIMARY KEY AUTO_INCREMENT, STestID = INT, TestID = INT, Score = INT, Answer = VARCHAR(500));
